    Ubuntu #

    apt # 

    sudo apt install

# download .deb package
sudo dpkg -i fd_8.0.0_amd64.deb  # adapt version number and architecture

    apt-get # 

    sudo apt-get install

    Fedora #

    dnf # 

    dnf install
